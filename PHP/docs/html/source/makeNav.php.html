function getActivePage(){
	$path = explode('/', parse_url($_SERVER['REQUEST_URI'])["path"]);
	if(end($path) == ""){
		unset($path[count($path)-1]);
	}
	if("/".end($path) == getSubfolder() || end($path) == getSubfolder()){
		return "home";
	}
	return end($path);
}

function getSubfolder(){
	$path = explode('/', parse_url($_SERVER['REQUEST_URI'])["path"]);
	unset($path[count($path)-1]);
	return implode($path, "/");
}
